CREATE OR REPLACE FUNCTION SUM_DAYS
(data_wypozyczenia DATE,data_zwrotu DATE)
RETURN NUMBER
IS
	sumaDNI NUMBER(7) := 0;
BEGIN
  SELECT DATEDIFF(day,data_wypozyczenia,data_zwrotu) AS sumaDNI;
	
	RETURN sumaDNI;
END;


CREATE OR REPLACE FUNCTION SUM_DAYS
(data_wypozyczenia IN DATE)
RETURN NUMBER
IS
	sumaDNI NUMBER(7) := 0;
BEGIN
  SELECT DATEDIFF(day,data_wypozyczenia,CURRENT_DATE) AS sumaDNI;
	
	RETURN sumaDNI;
END;
